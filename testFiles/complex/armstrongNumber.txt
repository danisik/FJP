    boolean test = isArmstrongNumber(153); // result is 1
    boolean test1 = isArmstrongNumber(150); // result is 0

    boolean function isArmstrongNumber(int n)
    {
        int c = a = temp = 0;

        temp = n;
        while(n > 0)
        {
            a = n % 10;
            n = n / 10;
            c = c + (a * a * a);
        }

        return temp == c;
    }
}
